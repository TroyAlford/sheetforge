import Layout from '@/models/Layout'
import Sheet from '@/components/Sheet'

## Character Sheet

<Sheet
  layout={Layout.create({
    children: [
      { children: [{ type: 'Descriptor' }] },
      { children: [{ type: 'Attribute' }] },
      { children: [
        { colSpan: 1, children: [{ type: 'Trait' }, { type: 'Resource' }] },
        { colSpan: 2, children: [{ type: 'Skill' }, { type: 'Item' }, { type: 'Spell' }] },
      ] },
    ]
  })}
  character={Character.create({})}
  onChange={(character, layout) => {
    localStorage.setItem('sf-character', JSON.stringify(character))
    localStorage.setItem('sf-layout', JSON.stringify(layout))
  }}
  onMount={(self) => {
    const character = localStorage.getItem('sf-character')
    if (character) { self.props.character.set(JSON.parse(character)) }
  }}
/>
